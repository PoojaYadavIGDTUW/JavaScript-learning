## 💻 Code Example

```javascript
let name = "Chiku";
console.log(name);

// var
var x = 10;
console.log(x);

// let
let y = 20;
console.log(y);

// const
const z = 30;
console.log(z);

// block scope
{
  let a = 10;
  const b = 20;
}
